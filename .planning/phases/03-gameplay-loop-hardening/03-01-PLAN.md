<!-- Source: src/app/api/user/route.ts -->
```typescript
import { NextRequest, NextResponse } from "next/server";
import { supabaseServer as supabase } from "@/lib/supabase-server";

function getUserId(req: NextRequest): string | null {
  const auth = req.headers.get("authorization");
  if (auth?.startsWith("Bearer ")) return auth.slice(7);
  return null;
}
```

<!-- Inventory query shape — use supabaseServer, not inventoryService (anon client). -->
<!-- Source: src/lib/services/inventoryService.ts query shape -->
```typescript
const { data, error } = await supabase
  .from("user_inventory")
  .select("*, items(*)")
  .eq("user_id", userId);

const equipped   = (data || []).filter((item: any) =>  item.equipped);
const unequipped = (data || []).filter((item: any) => !item.equipped);
return NextResponse.json({ equipped, unequipped });
```

<!-- Source: src/lib/services/shadowService.ts query shape -->
```typescript

  .select("*, shadows(*)")
  .eq("user_id", userId);
```
