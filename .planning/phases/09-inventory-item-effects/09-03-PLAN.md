```ts
const handleToggleEquip = async (item: UserItem) => {
  if (toggling) return;
  setToggling(true);
  systemAudio?.playClick();

  const newEquipStatus = !item.equipped;

  try {
    const res = await fetch("/api/inventory/equip", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${userId}`,
      },
      body: JSON.stringify({ userItemId: item.id, equip: newEquipStatus }),
    });
    const json = await res.json();
    if (!res.ok || !json.success) throw new Error(json.error || "equip failed");

    // Update local item list
    setItems(prev => prev.map(i => i.id === item.id ? { ...i, equipped: newEquipStatus } : i));
    setSelectedItem(prev => prev?.id === item.id ? { ...prev, equipped: newEquipStatus } : prev);

    // Fire stat delta notifications + patch game state
    const effects = item.items?.effects;
    if (effects) {
      const statKeys = ["strength", "vitality", "agility", "intelligence", "perception", "sense"] as const;
      const statPatch: Record<string, number> = {};

      for (const [key, val] of Object.entries(effects)) {
        if (statKeys.includes(key as any) && typeof val === "number") {
          const delta = newEquipStatus ? val : -val;
          statPatch[key] = delta;

          const label = key.toUpperCase();
          dispatch({
            type: "ADD_NOTIFICATION",
            payload: {
              type: "QUEST",
              title: newEquipStatus ? `${label} +${val}` : `${label} -${val}`,
              body: newEquipStatus ? `${item.items?.name} equipped` : `${item.items?.name} unequipped`,
              icon: newEquipStatus ? "⚔️" : "📦",
            },
          });
        }
      }

      // Patch stats in game state (additive delta via SET_DATA)
      // We don't have current stats here — Dashboard patches via callback
      // Strategy: dispatch ADD_STAT_BONUS action handled in reducer
      // Simple approach: dispatch SET_DATA with a stats delta flag is awkward.
      // Instead: pass onEquipChange callback to let Dashboard re-fetch inventory bonuses
      // See T2 for the callback wiring.
    }

    if (newEquipStatus) systemAudio?.playSuccess();
  } catch (err) {
    console.error("[Inventory] Equip failed:", err);
  } finally {
    setToggling(false);
  }
};
```

### T2 — Stats patch via `onEquipChange` callback (Dashboard-level re-merge)

**Rationale:** `Inventory.tsx` doesn't hold `state.stats`, so it can't compute new totals.
The cleanest approach is an `onEquipChange` prop that tells Dashboard to re-compute item
bonuses and patch the reducer.

**File:** `src/components/arise/Inventory.tsx`

Add prop:
```ts
interface InventoryProps {
  userId: string;
  dispatch: React.Dispatch<any>;
  onEquipChange?: () => void;
}
```

Call `onEquipChange?.()` after a successful equip/unequip.

**File:** `src/components/arise/Dashboard.tsx`

Replace `<Inventory userId={user.id} />` with:
```tsx
<Inventory
  userId={user.id}
  dispatch={dispatch}
  onEquipChange={async () => {
    const { getUserInventory } = await import("@/lib/services/inventoryService");
    const items = await getUserInventory(user.id);
    const bonuses = computeItemBonuses(items);
    dispatch({
      type: "SET_DATA",
      payload: {
        stats: { ...state.stats, ...mergeBonuses(state.user.stats, bonuses) },
      },
    });
  }}
/>
```

Define `computeItemBonuses` and `mergeBonuses` as local helpers (see T4).

### T4 — Wire inventory footer to live computed data

**File:** `src/components/arise/Inventory.tsx`

Replace the hardcoded footer array with computed values derived from `items` state:

```ts
// Compute footer values from equipped items
const equippedItems = items.filter(i => i.equipped);

// Active_Buffs: dominant bonus stat (highest single value across all equipped items)
const allBonuses: Array<{ stat: string; val: number }> = [];
for (const item of equippedItems) {
  if (item.items?.effects) {
    for (const [stat, val] of Object.entries(item.items.effects)) {
      if (["strength","vitality","agility","intelligence","perception","sense"].includes(stat)) {
        allBonuses.push({ stat, val: val as number });
      }
    }
  }
}
allBonuses.sort((a, b) => b.val - a.val);
const activeBuff = allBonuses.length > 0
  ? `${allBonuses[0].stat.toUpperCase()} +${allBonuses[0].val}`
  : "NONE";

// Defense_Rating: total VIT bonus from equipped items
const vitBonus = allBonuses.filter(b => b.stat === "vitality").reduce((s, b) => s + b.val, 0);
const defenseRating = vitBonus > 0 ? `VIT +${vitBonus}` : "0 BONUS";

// Global_Rarity: highest rarity among equipped items
const RARITY_ORDER = ["LEGENDARY", "EPIC", "RARE", "UNCOMMON", "COMMON"];
const highestRarity = equippedItems.reduce((best, item) => {
  const rank = RARITY_ORDER.indexOf(item.items?.rarity || "COMMON");
  const bestRank = RARITY_ORDER.indexOf(best);
  return rank < bestRank ? (item.items?.rarity || "COMMON") : best;
}, "COMMON");
```

### T5 — Merge item bonuses into stats at init (`page.tsx`)

**File:** `src/app/page.tsx`

After `loadUser` succeeds and before dispatching `SET_USER`, fetch inventory and merge
equipped item bonuses into base stats:

```ts
import { getUserInventory } from "@/lib/services/inventoryService";

// Helper (define above syncSession)
function computeItemBonuses(items: any[]): Partial<Record<string, number>> {
  const STAT_KEYS = ["strength","vitality","agility","intelligence","perception","sense"];
  const totals: Record<string, number> = {};
  for (const item of items) {
    if (!item.equipped || !item.items?.effects) continue;
    for (const [key, val] of Object.entries(item.items.effects as Record<string, number>)) {
      if (STAT_KEYS.includes(key)) totals[key] = (totals[key] ?? 0) + (val as number);
    }
  }
  return totals;
}
```

Inside `syncSession`, after `loadUser`:
```ts
const dbData = await loadUser(session.user.id);
if (dbData.user) {
  // Fetch inventory to compute equipped item bonuses
  const inventoryItems = await getUserInventory(session.user.id).catch(() => []);
  const itemBonuses = computeItemBonuses(inventoryItems);

  // Merge bonuses into base stats before dispatching
  const baseStats = mapDbUserToState(dbData.user, dbData.stats).stats;
  const mergedStats = baseStats ? {
    ...baseStats,
    strength:     (baseStats.strength     ?? 10) + (itemBonuses.strength     ?? 0),
    vitality:     (baseStats.vitality     ?? 10) + (itemBonuses.vitality     ?? 0),
    agility:      (baseStats.agility      ?? 10) + (itemBonuses.agility      ?? 0),
    intelligence: (baseStats.intelligence ?? 10) + (itemBonuses.intelligence ?? 0),
    perception:   (baseStats.perception   ?? 10) + (itemBonuses.perception   ?? 0),
    sense:        (baseStats.sense        ?? 10) + (itemBonuses.sense        ?? 0),
  } : baseStats;

  dispatch({
    type: "SET_USER",
    payload: { ...mapDbUserToState(dbData.user, dbData.stats), stats: mergedStats },
  });
  // ... rest of init (quests, penalty check)
}
```

## Acceptance Criteria

- Equipping an item dispatches `ADD_NOTIFICATION` per stat in its `effects` (STRENGTH +10 etc.)
- Unequipping fires matching negative notifications (STRENGTH -10)
- After equip/unequip, Dashboard stat numbers update (re-merge via `onEquipChange`)
- CONSUMABLE and SHADOW_FRAGMENT items show disabled USE [FUTURE] button, not EQUIP
- Footer: `Active_Buffs` shows real dominant bonus, `Defense_Rating` shows real VIT total, `Global_Rarity` shows highest rarity of equipped items
- On page load, equipped item bonuses are already merged into `state.stats`
- `toggleEquipItem` removed, anon-client write path eliminated
- TypeScript compiles without errors
